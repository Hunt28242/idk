Step 1: Go to script.google.com → New Project
Step 2: Delete everything and paste this code:
const SHEET_KEY = "jroc_coupons";
function doGet(e) {
  var props = PropertiesService.getScriptProperties();
  var action = e.parameter.action;
  var code = (e.parameter.code || "").toUpperCase();
  if (action === "create") {
    var obj = { label: e.parameter.label, amount: e.parameter.amount, used: false };
    props.setProperty(code, JSON.stringify(obj));
    return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);
  }
  if (action === "redeem") {
    var raw = props.getProperty(code);
    if (!raw) return ContentService.createTextOutput(JSON.stringify({ok:false,error:"invalid"})).setMimeType(ContentService.MimeType.JSON);
    var data = JSON.parse(raw);
    if (data.used) return ContentService.createTextOutput(JSON.stringify({ok:false,error:"used"})).setMimeType(ContentService.MimeType.JSON);
    data.used = true;
    props.setProperty(code, JSON.stringify(data));
    return ContentService.createTextOutput(JSON.stringify({ok:true,label:data.label,amount:data.amount})).setMimeType(ContentService.MimeType.JSON);
  }
  return ContentService.createTextOutput(JSON.stringify({ok:false,error:"unknown action"})).setMimeType(ContentService.MimeType.JSON);
}
Step 3: Click Deploy → New Deployment → Web App
